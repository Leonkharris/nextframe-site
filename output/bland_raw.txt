(function() {
  var STORAGE_KEY = 'bland_utm_data';
  var FIRST_SEEN_KEY = 'bland_first_pageview_seen';

  function getStoredUtms() {
    try {
      var raw = sessionStorage.getItem(STORAGE_KEY);
      return raw ? JSON.parse(raw) : null;
    } catch (e) {
      return null;
    }
  }

  function storeUtms(utms) {
    try {
      sessionStorage.setItem(STORAGE_KEY, JSON.stringify(utms));
    } catch (e) {}
  }

  function hasFirstPageBeenSeen() {
    try {
      return !!sessionStorage.getItem(FIRST_SEEN_KEY);
    } catch (e) {
      return false;
    }
  }

  function markFirstPageSeen() {
    try {
      sessionStorage.setItem(FIRST_SEEN_KEY, 'true');
    } catch (e) {}
  }

  function applyUtmsToUrl(utms) {
    if (!utms) return;

    var url = new URL(window.location.href);

    // Don’t overwrite if URL already has utm_source
    if (url.searchParams.has('utm_source')) {
      return;
    }

    Object.keys(utms).forEach(function(key) {
      if (utms[key]) {
        url.searchParams.set(key, utms[key]);
      }
    });

    history.replaceState(null, '', url.toString());
    console.log('UTMs applied to URL from sessionStorage:', utms);
  }

  // --- MAIN ---

  var url = new URL(window.location.href);

  // 1) If current URL already has UTMs, store them & done
  //    (e.g. Google Ads, Facebook Ads, or any manually tagged campaign)
  if (url.searchParams.has('utm_source')) {
    var utmsFromUrl = {
      utm_source:  url.searchParams.get('utm_source'),
      utm_medium:  url.searchParams.get('utm_medium'),
      utm_campaign: url.searchParams.get('utm_campaign'),
      utm_term:    url.searchParams.get('utm_term'),
      utm_content: url.searchParams.get('utm_content')
    };
    storeUtms(utmsFromUrl);
    console.log('UTMs found in URL; stored in session:', utmsFromUrl);
    return;
  }

  // 2) No UTMs in URL → see if we already have a stored source for this session
  var stored = getStoredUtms();
  if (stored) {
    // Persist same UTMs across pages
    applyUtmsToUrl(stored);
    return;
  }

  // 3) No UTMs anywhere yet → classify ONLY on first pageview
  if (!hasFirstPageBeenSeen()) {
    markFirstPageSeen();

    var ref = document.referrer;

    // Has a referrer (LinkedIn, YouTube, Reddit, Google, etc.)
    if (ref && ref !== '') {
      try {
        var refUrl = new URL(ref);
        var domain = refUrl.hostname.replace(/^www\./, '');

        // Auto-inferred sessions: only utm_source + utm_medium
        var inferredUtms = {
          utm_source: domain,       // e.g. linkedin.com, youtube.com, reddit.com
          utm_medium: 'referral'
        };

        storeUtms(inferredUtms);
        applyUtmsToUrl(inferredUtms);

        console.log('Referrer detected → UTMs set from referrer:', inferredUtms, 'referrer:', ref);
      } catch (e) {
        console.log('Referrer exists but URL parsing failed:', ref, e);
      }
      return;
    }

    // True direct traffic: no referrer and no UTMs
    // Direct should only have utm_source
    var directUtms = {
      utm_source: 'direct'
    };
    storeUtms(directUtms);
    applyUtmsToUrl(directUtms);
    console.log('Direct session detected; UTMs set to direct.');
  } else {
    console.log('Later pageview without stored UTMs; doing nothing.');
  }

  function shuffle(array) {
    const arr = array.slice();
    for (let i = arr.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [arr[i], arr[j]] = [arr[j], arr[i]];

    let currentIndex = 0;
    let autoplay = null;
    let isRunning = false;

    function resetAllFills() {
      links.forEach((link) => {
        const fill = link.querySelector(".workflow-line_fill");
        link.classList.remove("is-filling");

        if (fill) {
          fill.style.transition = "none";
          fill.style.width = "0%";
          fill.offsetHeight;
        }
      });
    }

    function startFill(activeLink) {
      const fill = activeLink.querySelector(".workflow-line_fill");
      if (!fill) return;

      activeLink.classList.remove("is-filling");
      fill.style.transition = "none";
      fill.style.width = "0%";
      fill.offsetHeight;

      activeLink.classList.add("is-filling");
      fill.style.transition = `width ${AUTO_ROTATE_DELAY}ms linear`;
      fill.style.width = "100%";
    }

    function activateTabByIndex(index, restartTimer = true) {
      currentIndex = index;

      links.forEach((link, i) => {
        const isActive = i === index;
        link.classList.toggle("w--current", isActive);
        link.setAttribute("aria-selected", isActive ? "true" : "false");
        link.tabIndex = isActive ? "0" : "-1";
      });

      panes.forEach((pane, i) => {
        pane.classList.toggle("w--tab-active", i === index);
      });

      resetAllFills();

      if (isRunning) {
        startFill(links[index]);
      }

      if (restartTimer && isRunning) {
        clearTimeout(autoplay);
        autoplay = setTimeout(() => {
          const nextIndex = (currentIndex + 1) % links.length;
          activateTabByIndex(nextIndex, true);
        }, AUTO_ROTATE_DELAY);
      }
    }

    function start() {
      if (isRunning) return;
      isRunning = true;

      resetAllFills();
      startFill(links[currentIndex]);

      clearTimeout(autoplay);
      autoplay = setTimeout(() => {
        const nextIndex = (currentIndex + 1) % links.length;
        activateTabByIndex(nextIndex, true);
      }, AUTO_ROTATE_DELAY);
    }

    function stop() {
      isRunning = false;
      clearTimeout(autoplay);
      resetAllFills();
    }

    const system = {
      holder,
      subsectionName,
      start,
      stop,
      activateTabByIndex
    };

    links.forEach((link, index) => {
      link.addEventListener("click", function (e) {
        e.preventDefault();

        if (activeSystem !== system) {
          if (activeSystem) activeSystem.stop();
          activeSystem = system;
          activeSystem.start();
        }

        activateTabByIndex(index, true);
      });
    });

    const initialIndex = links.findIndex((link) => link.classList.contains("w--current"));
    currentIndex = initialIndex >= 0 ? initialIndex : 0;
    activateTabByIndex(currentIndex, false);

    return system;
  }

  holders.forEach((holder) => {
    const system = createTabSystem(holder);
    if (system) tabSystems.push(system);
  });

  function setActiveSystem(nextSystem) {
    if (activeSystem === nextSystem) return;

    if (activeSystem) activeSystem.stop();
    activeSystem = nextSystem;
    if (activeSystem) activeSystem.start();
  }

  function detectActiveHolder() {
    const triggerY = window.innerHeight * 0.5;
    let matchedSystem = null;

    tabSystems.forEach((system) => {
      const rect = system.holder.getBoundingClientRect();
      const isAtMiddle = rect.top <= triggerY && rect.bottom >= triggerY;

      if (isAtMiddle) {
        matchedSystem = system;
      }
    });

    if (matchedSystem) {
      setActiveSystem(matchedSystem);
    }
  }

  window.addEventListener("scroll", detectActiveHolder, { passive: true });
  window.addEventListener("resize", detectActiveHolder);

  detectActiveHolder();

  if (!activeSystem && tabSystems.length) {
    setActiveSystem(tabSystems[0]);
  }
});
